import eyed3

def edit_music_message(music_path,music_title,artist,album,album_artist):
    """
    @param music_path: жӯҢжӣІж–Үд»¶и·Ҝеҫ„
    @param artist: жӯҢжӣІдҪңиҖ…
    @param album: дё“иҫ‘еҗҚз§°
    @param album_artist: дё“иҫ‘дҪңиҖ…
    @param music_title:  жӯҢжӣІеҗҚ
    """
    audiofile = eyed3.load(music_path) # еҠ иҪҪжӯҢжӣІж–Үд»¶
    audiofile.tag.title = music_title  # жӯҢжӣІеҗҚ
    audiofile.tag.artist = artist  # дҪңиҖ…
    audiofile.tag.album = album      # е”ұзүҮйӣҶ
    audiofile.tag.album_artist = album_artist  # дҪңиҖ…
    audiofile.tag.track_num = 3     #йҹіиҪЁеҸ·пјҢеңЁдё“иҫ‘дёӯжҺ’з¬¬еҮ 
    audiofile.tag.save() # дҝқеӯҳж–Үд»¶
    return 'жӯҢжӣІдҝЎжҒҜдҝ®ж”№жҲҗеҠҹ'
